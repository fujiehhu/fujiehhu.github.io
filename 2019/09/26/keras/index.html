keras 深度学习代码

import os
from keras import optimizers
from keras import applications
from keras.models import Sequential, Model
from keras.callbacks import ModelCheckpoint
from keras.layers import Dropout, Flatten, Dense
from keras.preprocessing.image import ImageDataGenerator
import matplotlib.pyplot as plt
from keras.callbacks import ReduceLROnPlateau
base_dir = '  '
train_dir = os.path.join(base_dir, 'train')
validation_dir = os.path.join(base_dir, 'validation')
batch_size = 20  # 批量大小
epochs = 40  # 迭代次数
img_height, img_width = 512, 512  # 图片高宽
train_data_dir = os.path.join(base_dir, 'train')  # 训练集目录
validation_data_dir = os.path.join(base_dir, 'validation')  # 测试集目录
OUT_CATEGORIES = 8  # 分类数
nb_train_samples =    # 训练样本数
nb_validation_samples =    # 验证样本数
# 定义模型
base_model = applications.VGG16(weights="imagenet", include_top=False,input_shape=(img_width, img_height,3))  
# 预训练的VGG19网络，替换掉顶部网络
top_model = Sequential()  # 自定义顶层网络
top_model.add(Flatten(input_shape=base_model.output_shape[1:]))  # 将预训练网络展平
top_model.add(Dense(512, activation='relu'))  # 全连接层，
top_model.add(Dropout(0.5))  # Dropout概率0.5
top_model.add(Dense(OUT_CATEGORIES, activation='softmax'))  # 输出层，二分类

model = Model(inputs=base_model.input, outputs=top_model(base_model.output))  # 新网络=预训练网络+自定义网络

model.compile(loss='sparse_categorical_crossentropy', optimizer=optimizers.SGD(lr=1e-3, momentum=0.9),
              metrics=['accuracy'])  # 损失函数为二进制交叉熵，优化器为SGD
#数据增强
train_datagen = ImageDataGenerator(
      zoom_range=[1.1,1.1],
      rescale=1./255,
      rotation_range=180,
      shear_range=1,
      horizontal_flip=True,
      vertical_flip=True,
      width_shift_range=0.35,
      height_shift_range=0.35,
      fill_mode = 'constant',
      cval = 0.0)


test_datagen = ImageDataGenerator(  rescale=1./255 )  # 测试数据预处理器

train_generator = train_datagen.flow_from_directory(train_data_dir, target_size=(img_height, img_width),
                                                    batch_size=batch_size, class_mode='binary')  # 训练数据生成器
validation_generator = test_datagen.flow_from_directory(validation_data_dir, target_size=(img_height, img_width),
                                                        batch_size=batch_size, class_mode='binary',
                                                        shuffle=False)  # 验证数据生成器
filepath='/content/drive/My Drive/vgg/model/VGG19-{epoch:03d}-{val_loss:.4f}-{val_acc:.4f}.h5'
checkpointer = ModelCheckpoint(filepath, monitor='val_acc',verbose=1, save_best_only=True)  # 保存最优模型
#降低学习率1
import keras.backend as K
from keras.callbacks import LearningRateScheduler

def scheduler(epoch):

    if epoch  == 8:
        lr = K.get_value(model.optimizer.lr)
        K.set_value(model.optimizer.lr, lr * 0.1)
        print("lr changed to {}".format(lr * 0.5))
    elif epoch==15:
        lr = K.get_value(model.optimizer.lr)
        K.set_value(model.optimizer.lr, lr * 0.1)
        print("lr changed to {}".format(lr * 0.8))
    elif epoch==20:
        lr = K.get_value(model.optimizer.lr)
        K.set_value(model.optimizer.lr, lr * 0.1)
        print("lr changed to {}".format(lr * 0.5))
    elif epoch==28:
        lr = K.get_value(model.optimizer.lr)
        K.set_value(model.optimizer.lr, lr * 0.1)
        print("lr changed to {}".format(lr * 0.8))
    return K.get_value(model.optimizer.lr)
reduce_lr = LearningRateScheduler(scheduler)
history = model.fit_generator(train_generator, 
                              steps_per_epoch=nb_train_samples // batch_size, 
                              epochs=epochs,
                              validation_data=validation_generator, 
                              validation_steps=nb_train_samples // batch_size,
                              verbose=1, workers=12, 
                              class_weight='auto',
                              callbacks=[checkpointer,reduce_lr])  # 每轮一行输出结果，最大进程12
